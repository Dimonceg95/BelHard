# функция принимает список чисел возвращает среднее арифметическое четных


def odd_average(a):
    odd_dict = []
    for i in a:
        if i % 2 == 0:
            odd_dict.append(i)
    average = sum(odd_dict) / len(odd_dict)
    return average

print(odd_average([1, 2, 3, 4, 6, 10, 14]))
